(%i1) f4s4(x4,x3,x2,x1,alpha,beta) := ( 1 + (x1^2 + x2^2 + x3^2 + x4^2)/ alpha ^2 )^(-beta);

(%i2) f4s3(x3,x2,x1,alpha,beta) := defint( f4s4(x4,x3,x2,x1,alpha,beta), x4, -inf, inf );

(%i3) f4s2(x2,x1,alpha,beta) := defint( f4s3(x3,x2,x1,alpha,beta), x3, -inf, inf );

(%i4) f4s1(x1,alpha,beta) := defint( f4s2(x2,x1,alpha,beta), x2, -inf, inf );

(%i5) f4s0(alpha,beta) := defint( f4s1(x1,alpha,beta), x1, -inf, inf );

(%i6) norm(alpha, beta) := factor(makegamma(ratsimp(f4s0(alpha,beta))));

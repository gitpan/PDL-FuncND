(%i1) f3s3(x3,x2,x1,alpha,beta) := ( 1 + (x1^2 + x2^2 +x3^2)/ alpha ^2 )^(-beta);

(%i2) f3s2(x2,x1,alpha,beta) := defint( f3s3(x3,x2,x1,alpha,beta), x3, -inf, inf );

(%i3) f3s1(x1,alpha,beta) := defint( f3s2(x2,x1,alpha,beta), x2, -inf, inf );

(%i4) f3s0(alpha,beta) := defint( f3s1(x1,alpha,beta), x1, -inf, inf );

(%i5) norm(alpha, beta) := factor(makegamma(ratsimp(f3s0(alpha,beta))));

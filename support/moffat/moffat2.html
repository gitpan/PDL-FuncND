Simplified 2D Moffat

(%i1) f2s2(x2,x1,alpha,beta) := ( 1 + (x1^2 + x2^2)/ alpha ^2 )^(-beta);

(%i2) f2s1(x1,alpha,beta) := defint( f2s2(x2,x1,alpha,beta), x2, -inf, inf );

(%i3) f2s0(alpha,beta) := defint( f2s1(x1,alpha,beta), x1, -inf, inf );

(%i4) norm(alpha, beta) := factor(makegamma(ratsimp(f2s0(alpha,beta))));

(%i5) killcontext(initial); assume( alpha > 0, beta > 0, beta < 0.5 ); norm(alpha,beta);
